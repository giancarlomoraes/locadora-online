create table categoria(
	id serial primary key,
	nome varchar (20)
);

create table filme(
	id serial primary key,
	id_categoria integer not null,
	titulo varchar(50),
	duracao varchar(20),
	constraint fk_filme_categoria foreign key (id_categoria) references categoria(id)

);

create table cliente(
	id serial primary key,
	prenome varchar(20),
	sobrenome varchar(50),
	email varchar(50),
	telefone varchar(11),
	endereco varchar(100)
);

create table fita(
	id serial primary key,
	id_cliente integer not null,
	id_filme integer not null,
	constraint fk_fita_cliente foreign key (id_cliente) references cliente(id),
	constraint fk_fita_filme foreign key (id_filme) references filme(id)

);
